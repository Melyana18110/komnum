    f'(x) = (2e)^2x f'(0) = 2

    f”(x) = (4e)^2x f”(0) = 4

    f”'(x) = (8e)^2x f”'(0) = 8

  • Program

import math

x=4
cek =1
a = 0
b=1
iter = 1

while cek>0.001:
    f_x = 0
    f_y = 0
    for i in range(a):
        f_x += (2**i)*x**i/math.factorial(i)

    for j in range(b):
        f_y += (2**j)*x**j/math.factorial(j)
    cek = f_y-f_x
    a+=1
    b+=1
    print("iterasi ke-",iter,"= ",cek)
    iter+=1
